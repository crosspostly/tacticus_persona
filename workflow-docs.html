def sort_data_json():
    """–°–æ—Ä—Ç–∏—Ä—É–µ—Ç characters –≤ data.json –ø–æ –∏–º–µ–Ω–∏"""
    
    # –ß–∏—Ç–∞–µ–º —Ñ–∞–π–ª
